#!/bin/bash
# sync_conflict_resolver.sh - Automated conflict detection and resolution helper

VAULT_PATH="${1:-$HOME/Obsidian/Vault}"
MODE="${2:-interactive}"  # interactive | keep-main | keep-newest

echo "â•”â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•—"
echo "â•‘                  SYNC CONFLICT RESOLVER                      â•‘"
echo "â•šâ•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•"

# Find all conflicts
CONFLICTS=$(find "$VAULT_PATH" -name "*sync-conflict*" -type f 2>/dev/null)
CONFLICT_COUNT=$(echo "$CONFLICTS" | grep -c "sync-conflict" || echo 0)

if [ "$CONFLICT_COUNT" -eq 0 ]; then
    echo ""
    echo "âœ“ No sync conflicts found in $VAULT_PATH"
    exit 0
fi

echo ""
echo "Found $CONFLICT_COUNT sync conflict(s):"
echo ""

# Group conflicts by original file
declare -A CONFLICT_GROUPS

while IFS= read -r conflict; do
    # Extract original filename
    original=$(echo "$conflict" | sed 's/.sync-conflict-[0-9T-]*-[A-Z0-9]*\.md$/.md/')
    CONFLICT_GROUPS["$original"]+="$conflict|"
done <<< "$CONFLICTS"

# Process each group
for original in "${!CONFLICT_GROUPS[@]}"; do
    echo "â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€"
    echo "Original: $(basename "$original")"
    echo "Path: $original"
    echo ""
    
    # List conflict files
    IFS='|' read -ra conflicts <<< "${CONFLICT_GROUPS[$original]}"
    for c in "${conflicts[@]}"; do
        [ -z "$c" ] && continue
        # Cross-platform stat
        if [[ "$OSTYPE" == "darwin"* ]]; then
            mod_time=$(stat -f "%Sm" -t "%Y-%m-%d %H:%M:%S" "$c" 2>/dev/null)
        else
            mod_time=$(stat -c "%y" "$c" 2>/dev/null | cut -d'.' -f1)
        fi
        echo "  Conflict: $(basename "$c")"
        echo "  Modified: $mod_time"
        echo ""
    done
    
    if [ "$MODE" == "interactive" ]; then
        echo "Action: [k]eep main, [n]ewest wins, [d]iff, [s]kip"
        read -r -n1 action
        echo ""
        
        case $action in
            k)
                for c in "${conflicts[@]}"; do
                    [ -n "$c" ] && rm -v "$c"
                done
                ;;
            n)
                newest="$original"
                newest_time=0
                for c in "${conflicts[@]}"; do
                    [ -z "$c" ] && continue
                    # Cross-platform stat
                    if [[ "$OSTYPE" == "darwin"* ]]; then
                        c_time=$(stat -f "%m" "$c" 2>/dev/null)
                    else
                        c_time=$(stat -c "%Y" "$c" 2>/dev/null)
                    fi
                    if [ "$c_time" -gt "$newest_time" ]; then
                        newest="$c"
                        newest_time=$c_time
                    fi
                done
                if [ "$newest" != "$original" ]; then
                    mv -v "$newest" "$original"
                fi
                for c in "${conflicts[@]}"; do
                    [ -n "$c" ] && [ -f "$c" ] && rm -v "$c"
                done
                ;;
            d)
                for c in "${conflicts[@]}"; do
                    [ -n "$c" ] && diff -y "$original" "$c" | head -50
                done
                ;;
            *)
                echo "Skipped"
                ;;
        esac
    fi
done

echo ""
echo "â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€"
REMAINING=$(find "$VAULT_PATH" -name "*sync-conflict*" -type f 2>/dev/null | wc -l | tr -d ' ')
echo "Remaining conflicts: $REMAINING"

    for link in $outlinks; do
        # Record outlink
        OUTLINKS["$filename"]+="$link|"
        # Record inlink for target
        INLINKS["$link"]+="$filename|"
    done

    filename=$(basename "$file" .md)

DRAFTS=()
DAILY=()
UNTITLED=()
OTHER=()

    name=$(basename "$orphan")
    if [[ "$name" =~ ^[0-9]{4}-[0-9]{2}-[0-9]{2} ]]; then
        DAILY+=("$orphan")
    elif [[ "$name" =~ ^Untitled ]]; then
        UNTITLED+=("$orphan")
    elif [[ "$name" =~ ^Draft|^WIP|^TODO ]]; then
        DRAFTS+=("$orphan")
    else
        OTHER+=("$orphan")
    fi

printf "  %-20s %d\n" "Untitled notes:" "${#UNTITLED[@]}"

# Show sample

for orphan in "${ORPHANS[@]:0:10}"; do
    echo "  â€¢ $(basename "$orphan")"
done

echo ""
echo "â•â•â• RECOMMENDATIONS â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•"
echo ""
echo "  1. Create MOC (Map of Content) note linking related orphans"
echo "  2. Add orphans to relevant daily notes with [[link]]"

echo "  4. Use Dataview to auto-list orphans:"
